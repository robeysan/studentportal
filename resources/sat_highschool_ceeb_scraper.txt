for i in AL AK AZ AR CA CO CT DE FL GA HI ID IL IN IA KS KY LA ME MD MA MI MN MS MO MT NE NV NH NJ NM NY NC ND OH OK OR PA RI SC SD TN TX UT VT VA WA WV WI WY; do wget "http://sat.collegeboard.org/register/sat-code-search-schools?decorator=none&submissionMode=ajax&pageId=registerCodeSearch&codeType=high-school-code&country=US&state=$i&city=" -q -O $i.html; grep '<td' $i.html | sed -e 's/<td class=.*">//g; s/ &nbsp;<\/td>//;' | tr '\r' ' ' | while read school && read id; do echo "{\"name\":\"$school\",\"ceeb\":$id,\"state\":\"$i\"}"; done > $i.json; done && rm states.json && cat *.json > states.json
